<VirtualHost *:8000>
    ServerAdmin webmaster@localhost
    DocumentRoot /app

    Header always set Cache-Control "no-cache, no-store, must-revalidate"
    Header always set Pragma "no-cache"
    Header always set Expires "0"

    Alias /static /app/static
    <Directory /app/static>
        Require all granted
        Allow from all
    </Directory>

    <Directory /app/cloudunderroof>
        <Files wsgi.py>
            Require all granted
            Allow from all
        </Files>
    </Directory>

    #WSGIDaemonProcess cloudunderroof python-path=/usr/local python-home=/usr/local/
    WSGIDaemonProcess cloudunderroof \
        python-path=/app:/usr/local/lib/python3.13/site-packages \
        processes=2 threads=5

    WSGIProcessGroup cloudunderroof
    WSGIScriptAlias / /app/cloudunderroof/wsgi.py

    ErrorLog ${APACHE_LOG_DIR}/cloudunderroof_error.log
    CustomLog ${APACHE_LOG_DIR}/cloudunderroof_access.log combined
</VirtualHost>
